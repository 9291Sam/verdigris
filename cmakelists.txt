cmake_minimum_required(VERSION 3.28)
project(verdigris)

include(FetchContent)

set(CMAKE_CXX_STANDARD 23)

FetchContent_Declare(fmt
        GIT_REPOSITORY https://github.com/fmtlib/fmt.git
        GIT_TAG master
        GIT_SHALLOW TRUE
        )
FetchContent_MakeAvailable(fmt)

FetchContent_Declare(concurrentqueue
        GIT_REPOSITORY https://github.com/cameron314/concurrentqueue
        GIT_TAG master
        GIT_SHALLOW TRUE
        )
FetchContent_MakeAvailable(concurrentqueue)

add_executable(verdigris
  src/main.cpp

  src/gfx/renderer.cpp

  src/util/log.cpp
  src/util/misc.cpp
)
target_compile_definitions(verdigris PUBLIC VERDIGRIS_ENABLE_VALIDATION)
target_link_libraries(verdigris PUBLIC fmt::fmt concurrentqueue)
target_include_directories(verdigris PUBLIC ${CMAKE_SOURCE_DIR}/src)
